package com.company;
import java.util.Vector;

public class Stack {
    private Vector vec = new Vector();

    public boolean empty(){
        return vec.isEmpty();
    }

    public void push(Object element){
        vec.add(element);
    }

    public int peek(){
        int element;
        element =  Integer.valueOf((int)vec.firstElement());
        return element;
    }
    
    public int pop(){
        int element;
        element =  Integer.valueOf((int)vec.firstElement());
        vec.remove(0);
        return element;
    }
}
